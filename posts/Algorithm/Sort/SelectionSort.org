  def selectionSort(arr):
      """
      指针i用于遍历数组，每次从 i + 1的位置开始往后找，把最小的数与i交换位置
      时间复杂度始终为 O(n^2)
      """
      n = len(arr)
      for i in range(n):
          # 假设当前位置是最小值的位置
          min_idx = i
          # 从i+1位置到结尾寻找比当前位置小的元素
          for j in range(i+1, n):
              if arr[j] < arr[min_idx]:
                  min_idx = j
          # 将找到的最小值与当前位置的值交换
          arr[i], arr[min_idx] = arr[min_idx], arr[i]
      return arr
