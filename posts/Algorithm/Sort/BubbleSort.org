  def bubbleSort(arr):
      n = len(arr)
      # 遍历所有数组元素
      for i in range(n):
          # 标记变量，用于优化
          swapped = False
          # 最后i个元素已经到位，无需比较
          for j in range(0, n-i-1):
              # 遍历数组从0到n-i-1，交换如果元素找到比它大的元素
              if arr[j] > arr[j+1]:
                  arr[j], arr[j+1] = arr[j+1], arr[j]
                  swapped = True
          # 如果在内部循环中没有交换，则数组已经排序完成
          if not swapped:
              break
      return arr
